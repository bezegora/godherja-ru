report_aervalr_shard = {
	type = alert
	check_create_action = {
		if = {
			limit = {
				has_variable = recent_aervalr_shard
			}
			var:recent_aervalr_shard = {
				save_scope_as = recent_aervalr_shard_county
			}
			try_create_important_action = {
				important_action_type = report_aervalr_shard
				actor = root
			}
		}
	}
	effect = {
		pan_camera_to_title = scope:recent_aervalr_shard_county
	}
}
