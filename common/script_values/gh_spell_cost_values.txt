	value = 0
	if = {	# The closer you beat yourself to the brink of death, the more magic you get
		limit = { 
			NOT = { 
				has_trait_rank = {
					trait = wounded
					rank >= 1
				}
			} 
		}
		add = tier_2_gen
	}
	else_if = {
		limit = { 
			has_trait_rank = {
				trait = wounded
				rank = 1
			}
		}
		add = tier_3_gen
	}
	else_if = {
		limit = { 
			has_trait_rank = {

				rank = 2
			}
		}
		add = tier_4_gen
	}
	if = {
		limit = { 
			has_perk = livingmagic_perk3_tree1
		}
		multiply = 1.5
	}
