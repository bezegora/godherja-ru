culture_ai_weight_skirmishers = {
	value = -10
	
	if = { # Cultures that favours skirmishers
		limit = {
			OR = {
				culture_group = culture_group:gh_sjalvolki_group
				culture_group = culture_group:gh_goans_group
				culture_group = culture_group:gh_allemani_group
				culture_group = culture_group:gh_gallieni_group
				culture_group = culture_group:gh_fog_eater_group
				culture_group = culture_group:gh_amharian_group
				culture_group = culture_group:gh_mondecny_group
				culture_group = culture_group:gh_kikarsag_group
				culture = culture:centralimperial
				culture = culture:islander
				culture = culture:centralislander
				culture_group = culture_group:gh_sarridian_group
				culture_group = culture_group:gh_partic_group
				culture_group = culture_group:gh_sarminian_group
				culture_group = culture_group:gh_karzarads_group
				culture_group = culture_group:gh_huegodocs_group
			}
		}
		add = 30
	}
	if = { # Cultures that dislike skirmishers
		limit = {
			OR = {
				culture_group = culture_group:gh_marcher_group
			}
		}
		add = -60
	}
	
	if = { # Cultures with a special MaA of that type should avoid the base version of the unit
		limit = {
			culture = {
				OR = {
					has_innovation = innovation_berserkers
					has_innovation = innovation_moonkin
					has_innovation = innovation_wodes
					has_innovation = innovation_eastern_cav
					has_innovation = innovation_bloodbound
					has_innovation = innovation_legion_raiders
					has_innovation = innovation_legion_marines
				}
			}
		}
		add = -100
	}
}
