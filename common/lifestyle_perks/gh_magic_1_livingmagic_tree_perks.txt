	icon = node_martial

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_magic_trigger = yes
			}
			add = 1989
		}
		if = {

		}

			}

		}

			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}

		}
	}

	can_be_picked = {
		has_education_living_magic_trigger =  yes
	}

	effect = {
		custom_tooltip = lm_perk1.tt
	}
