deadmagic_perk1_tree1 = {
	lifestyle = magic_lifestyle_dead
	tree = deadmagic_tree1
	position = { 1 0 }
	icon = node_intrigue
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_magic_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = magic_power_focus_dead
			}
			multiply = 5
		}
		if = {
			limit = {
				has_education_dead_magic_trigger = yes
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}
	
	can_be_picked = {
		has_education_dead_magic_trigger = yes
	}
	
	effect = {
		custom_description_no_bullet = {
			text = deadmagic_perk1_effect
		}
	}
	
	character_modifier = {
		dread_baseline_add = 5
	}
}
