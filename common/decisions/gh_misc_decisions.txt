found_metro_in_kingdom_cap = {
	picture = "gfx/interface/illustrations/decisions/decision_misc.dds"
	major = yes
    
	ai_check_interval = 72
    ai_potential = {
		always = yes
	}
    ai_will_do = 1
    
	desc = found_metro_in_kingdom_cap_desc 
	selection_tooltip = redistrict_metro_cap_tooltip
	is_shown = {
		capital_province = {
			is_any_metropolis_district = no
		}
		any_held_title = {
			count >= 1
			tier >= tier_kingdom
		}
	}
	is_valid = {
		any_held_title = {
			count >= 1 
			tier >= tier_kingdom 
		}
		prestige_level >= 4
        or = {
            diplomacy >= 20
            martial >= 20
            stewardship >= 20
            intrigue >= 20
            learning >= 20
        }
	}
	effect = {
		trigger_event = gh_misc_decision_events.002
	}
	cost = {
		gold = {
            value = 1500
        }
	}
}

####debasing_currency_decisions
gh_debase_currency_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"

	desc = gh_debase_currency_decision_desc

	selection_tooltip = gh_debase_currency_decision_tooltip

	major = yes
	
	is_shown = {
		is_ruler = yes
		is_landed = yes
		capital_county = {
            any_county_province = {
               has_building = mint_01
            }
        }
	}

	is_valid = {
        is_ruler = yes
        is_landed = yes
        capital_county = {
            any_county_province = {
               has_building = mint_01
            }
        }    
		NOT = {
			has_character_modifier = gh_strengthen_currency_modifier
		}
    }
	
	cost = {
		prestige = 500
	}

	effect = {
		add_gold = 300
		add_character_modifier = {
			modifier = gh_debased_currency_modifier
			years = 10
		}
	}
	
	cooldown = { years = 10 }
	
	ai_check_interval = 72
	
	ai_potential = {
		ai_greed >= medium_positive_ai_value
		ai_honor <= medium_negative_ai_value
		gold < medium_gold_value
	}

	ai_will_do = {
		base = 10
		
		modifier = {
			add = 45
			ai_greed >= high_positive_ai_value
		}
		modifier = {
			add = 45
			ai_honor <= high_negative_ai_value
		}
	}
}
