migrate_to_kingdom_berlisk = {
	picture = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	major = yes
	desc = migrate_to_kingdom_desc
    confirm_text = migrate_to_kingdom_confirm

	ai_check_interval = 1

	is_shown = {
		primary_title = title:e_sjalvolki
		has_culture_group = culture_group:gh_sjalvolki_group
		is_landed = yes
		has_game_rule = story_content_enabled
		is_target_in_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:sjaklaki_migration
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:berlisk_migration
			}
		}
	}

	is_valid = {
		completely_controls = title:d_hyblossos
		completely_controls = title:d_myonassus
		completely_controls = title:d_gorteutherna
		completely_controls = title:d_helleia
		completely_controls = title:d_platamum
		completely_controls = title:d_thermyra
		completely_controls = title:d_eleutidrus
		completely_controls = title:d_pylynthus
		completely_controls = title:d_naxithos
	}

	effect = {
		trigger_event = sjalvolki_decisions.3000
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:berlisk_migration
		}
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 60
	}
}
