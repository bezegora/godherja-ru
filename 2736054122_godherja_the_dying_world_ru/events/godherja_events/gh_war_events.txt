#Event for the execution of Aeschreas if he loses a war
war_event.3104 = {
	
	title = war_event.3104.t
	desc = war_event.3104.desc

	theme = prison
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = character:western_imperial_1
		animation = rage
	}

	immediate = { 
		character:western_imperial_1 = {
			save_scope_as = prisoner_a
		}
	}

	 option = {
		name = war_event.3104.a    #Execute Aeschreas

		scope:prisoner_a = {
			death = {
				death_reason = death_execution
				killer = root
			}
		}
		add_trait = gh_aeschreas_killer
		add_piety = monumental_piety_gain
		add_prestige = monumental_prestige_gain
		if = {
			limit = { has_nickname = no }
			give_nickname = nick_archon_killer
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = war_event.3104.b
		ai_chance = {
			base = 0
		}
	}
}
