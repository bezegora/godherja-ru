agionist.0001 = {
	type = empty
	hidden = yes
	
	immediate = {
		title:k_saint_sansias_rose = {
			holder = {
				every_vassal_or_below = {
					limit = {
						OR = {
							has_trait = compassionate
							has_trait = forgiving
							has_trait = humble
						}
						NOT = { has_variable = good_agionist }
					}
					if = {
						limit = { has_trait = compassionate }
						remove_trait = compassionate
						add_trait = sadistic
					}
					if = {
						limit = { has_trait = forgiving }
						remove_trait = forgiving
						add_trait = vengeful
					}
				}
			}
		}
	}
}
