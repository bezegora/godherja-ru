# Sets whether the defendant is actually guilty, how much evidence exists against them and the player's intuitive guess on this
scripted_effect determine_criminal_case_background_effect = {
	# First, determine whether the defendant is actually guilty or not
	random = {
		chance = 75
		save_scope_value_as = { 
			name = defendant_is_guilty 
			value = yes 
		}
	}
	# Determine the background of the defendant, for flavor purposes
	random_list = {
		70 = {
			set_variable = {
				name = defendant_background
				value = flag:peasant
				days = 1
			}
		}
		20 = {
			set_variable = {
				name = defendant_background
				value = flag:well_off_citizen
				days = 1
			}
		}
		10 = {
			set_variable = {
				name = defendant_background
				value = flag:petty_noble
				days = 1
			}
		}
	}
	# Then, determine the amount of evidence that exists against them
	random_list = {
		1 = {	# Practically no evidence
			set_variable = {
				name = evidence_level 
				value = 0 
				days = 1
			}
		}
		2 = {	# Little evidence
			set_variable = { 
				name = evidence_level 
				value = 25
				days = 1
			}
		}
		3 = {	# A medium amount of evidence
			set_variable = { 
				name = evidence_level 
				value = 50
				days = 1
			}
		}
		2 = {	# Significant evidence
			set_variable = { 
				name = evidence_level 
				value = 75
				days = 1
			}
		}
		1 = {	# Almost ironclad evidence
			set_variable = { 
				name = evidence_level 
				value = 100 
				days = 1
			}
		}
	}
	# Finally, determine the character's intuitive guess on the defendant's guilt
	set_variable = {
		name = criminal_case_intuitive_guess
		value = {
			value = var:evidence_level
			add = {
				integer_range = {
					min = -20
					max = 20
				}
			}
			if = {
				limit = { exists = scope:defendant_is_guilty }
				add = {
					integer_range = {
						min = root.intrigue
						max = {
							value = root.intrigue
							multiply = 2
						}
					}
				}
			}
			else = {
				subtract = {
					integer_range = {
						min = root.intrigue
						max = {
							value = root.intrigue
							multiply = 2
						}
					}
				}
			}
			# Paranoid and trusting characters have inherent biases
			if = {
				limit = { root = { has_trait = paranoid } }
				add = 10
			}
			else_if = {
				limit = { root = { has_trait = trusting } }
				subtract = 10
			}
			min = 0
			max = 100
		}
		days = 1
	}
}

scripted_effect execute_criminal_case_verdict_effect = {
	
	random_list = {
		50 = {
			modifier = {
				add = {
					value = var:evidence_level
					subtract = 50
					min = -49
				}
				NAND = {
					exists = scope:convicting
					var:criminal_case_intuitive_guess = { compare_value > root.var:evidence_level }
				}
			}
			modifier = {
				add = {
					value = var:criminal_case_intuitive_guess
					subtract = 50
					min = -49
				}
				exists = scope:convicting
				var:criminal_case_intuitive_guess = { compare_value > root.var:evidence_level }
			}
			modifier = {
				add = root.judicial_power_value
				exists = scope:convicting
			}
			modifier = {
				add = root.intrigue
				exists = scope:convicting
			}
			if = {
				limit = { exists = scope:convicting }
				add_piety = {
					value = {
						value = miniscule_piety_value
						multiply = criminal_case_guilty_difficulty_factor_value
					}
				}
				add_jurist_progress_effect = { VALUE = medium_jurist_progress_value }
				custom_tooltip = criminal_case_only_correct_judgement_will_give_jurist_progress_tooltip
			}
			else = {
				add_piety = miniscule_piety_loss
			}
			desc = criminal_case_conviction_seen_as_correct_tooltip
		}
		50 = {
			modifier = {
				add = {
					value = 50
					subtract = var:evidence_level
					min = -49
				}
				NAND = {
					exists = scope:acquitting
					var:criminal_case_intuitive_guess = { compare_value < root.var:evidence_level }
				}
			}
			modifier = {
				add = {
					value = 50
					subtract = var:criminal_case_intuitive_guess
					min = -49
				}
				exists = scope:acquitting
				var:criminal_case_intuitive_guess = { compare_value < root.var:evidence_level }
			}
			modifier = {
				add = root.judicial_power_value
				exists = scope:acquitting
			}
			modifier = {
				add = root.intrigue
				exists = scope:acquitting
			}
			if = {
				limit = { exists = scope:acquitting }
				add_piety = {
					value = {
						value = miniscule_piety_value
						multiply = criminal_case_acquittal_difficulty_factor_value
					}
				}
				add_jurist_progress_effect = { VALUE = medium_jurist_progress_value }
				custom_tooltip = criminal_case_only_correct_judgement_will_give_jurist_progress_tooltip
			}
			else = {
				add_piety = miniscule_piety_loss
			}
			desc = criminal_case_acquittal_seen_as_correct_tooltip
		}
	}
	if = {
		limit = {
			has_variable = judicial_approach
			var:judicial_approach = flag:innovation_approach
		}
		random = {
			chance = 25
			add_radicalism_effect = { VALUE = 15 }
		}
	}
}

		if = {
			limit = { exists = scope:high_court_scope }
			scope:high_court_scope = {
				fill_high_court_seat_with_character_effect = { JUDGE = scope:corrupt_judge }
			}
		}
		if = {
			limit = { exists = local_var:corruption_perk }
			if = {
				limit = { local_var:corruption_perk = flag:money }
				add_gold = local_var:money_reward
			}
			if = {
				limit = { local_var:corruption_perk = flag:blackmail }
				random_list = {
					80 = {
						add_hook = {
							type = favor_hook
							target = scope:vassal_to_blackmail
						}
					}
					20 = {
						add_hook = {
							type = fabrication_hook
							target = scope:vassal_to_blackmail
						}
					}
				}
			}
		}
		duel = {
			skill = intrigue
			value = decent_skill_rating
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				add_corruption_effect = { VALUE = scope:corrupt_judge.corruption_value }
			}
		}
	}
	
	option = {
		name = legal_tradition.1003.b
		add_prestige = major_prestige_loss
