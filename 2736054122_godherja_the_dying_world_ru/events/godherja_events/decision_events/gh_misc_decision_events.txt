gh_misc_decision_events.002 = {
	type = character_event
	title = gh_misc_decision_events.002.t
	desc = gh_misc_decision_events.002.desc
	theme = learning
	left_portrait = {
		character = ROOT 
		animation = personality_rational
	}
    
    immediate = { #saving scope for localization
        capital_province = {
            save_scope_as = capital
        }
    }

	#high martial options
	option = {
		name = gh_misc_decision_events.002.a
		trigger = {
			ROOT = {
				martial >= 20
			}
		}
		trigger_event = gh_misc_decision_events.003
        
        ai_chance = {
            base = 100
            ai_value_modifier = {
                ai_boldness = 1
                ai_greed = -0.2
            }
        }
	}
	#high steward options
	option = {
		name = gh_misc_decision_events.002.b
		trigger = {
			stewardship >= 20 
		}
		trigger_event = gh_misc_decision_events.004
        
        ai_chance = {
            base = 100
            ai_value_modifier = {
                ai_rationality = 1
            }
        }
	}
	#high diplomacy options
	option = {
		name = gh_misc_decision_events.002.c
		trigger = {
			diplomacy >= 20
		}
		trigger_event = gh_misc_decision_events.005
        
        ai_chance = {
            base = 100
            ai_value_modifier = {
                ai_honor = 1
                ai_greed = -0.2
            }
        }
	}
	#high intrigue options
	option = {
		name = gh_misc_decision_events.002.d
		trigger = {
			intrigue >= 20
		}
		trigger_event = gh_misc_decision_events.006
        
        ai_chance = {
            base = 100
            ai_value_modifier = {
                ai_honor = -1
            }
        }
	}
	#high learning options
	option = {
		name = gh_misc_decision_events.002.e
		trigger = {
			learning >= 20
		}
		trigger_event = gh_misc_decision_events.007
	}
    
    ai_chance = {
        base = 100
        ai_value_modifier = {
         	ai_zeal = 1
		 	ai_greed = -0.2
        }
    }
}

	#	trigger = {
	#		has_trait = architect
	#	}
	#}
	#lifestyle 2
	option = { #administrator - urban
		name = gh_misc_decision_events.004.c
		trigger = {
			has_trait = administrator
		}
		capital_province = {
			set_holding_type = urban_district_holding
		}
	}
	#lifestyle 3
	option = { #avaricious -trade/maratime
		name = gh_misc_decision_events.004.d
		trigger = {
			has_trait = avaricious
		}
		if = {
			limit = {
				capital_province = {
					is_coastal = yes 
				}
			}
			capital_province = {
				set_holding_type = maritime_district_holding
			}
		}
		else_if = {
			capital_province = {
				set_holding_type = trade_district_holding
			}
		}
	}
	#go back
	option = {
		name = gh_misc_decision_events.004.e
		trigger_event = gh_misc_decision_events.002	
	}	
}
