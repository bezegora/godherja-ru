notable.0002 = { # Search for a military councilor

	left_portrait = scope:high_martial_skill_option

	trigger = { exists = capital_province } # Can't recruit anyone if you lose your capital before you get the event

	immediate = {
		hidden_effect = {
			#Create our councilors

			if = { #Performance heavy option finding is only for players
				limit = { is_ai = no }

					limit = {
						OR = {
							martial >= high_skill_rating

							location = root.capital_province

	                            }

						    add_trait = strategist

		                    count = { 0 1 }

		                    wrathful = {}
		                    zealous = {}
		                    impatient = {}

		trigger = {
			exists = scope:excellent_martial_skill_option

			target = scope:excellent_martial_skill_option

			reverse_add_opinion = {
