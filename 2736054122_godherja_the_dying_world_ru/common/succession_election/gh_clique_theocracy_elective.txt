clique_theocracy_elective = {
	elector_vote_strength = {
		base = 0
		modifier = { # Political power
			desc = tooltip_clique_political_power_elector_strength
			add = character_influence_total
		}
		
	}
	
	electors = {
		add = holder
		add = holder_direct_vassals
		add = title_claimants
		limit = { faith = scope:holder.faith }
	}

	candidate_score = {
		base = 0
		######################	Elector self-voting pattern	##########################
		elector_self_voting_pattern_clique_elective_modifier = yes
		# some stuff about children of the ruler voting for themselves needed to be changed from the feudal one

		##########################	Elector voting patterns (circumstances)	##########################
		elector_voting_pattern_circumstances_feudal_elective_modifier = yes
		# doesn't need to be changed from the feudal one

		##########################	Elector voting patterns (prestige/piety)	##########################
		elector_voting_pattern_prestige_piety_holy_order_elective_modifier = yes
		# theocracies care a lot about piety, but not much about dynastic prestige

		##########################	Elector voting patterns (traits)	##########################
		elector_voting_pattern_traits_feudal_elective_modifier = yes
		# doesn't seem necessary to change this from the feudal one

		##########################	Elector voting patterns (opinion)	##########################
		elector_voting_pattern_opinion_feudal_elective_modifier = yes
		# doesn't seem necessary to change this from the feudal one either
		
		##########################	Elector voting patterns (cliques)	##########################
		elector_voting_pattern_circumstances_clique_mechanics_modifier = yes
	}
	
	#scope:candidate = Candidate, scope:title = Title, scope:holder_candidate = Candidate currently selected by ruler.
	candidates = {
		add = {
			type = holder_direct_vassals
			limit = {
				has_government = theocracy_government
				is_capable_adult = yes
				faith = scope:holder.faith
			}
		}
		add = {
			type = title_claimants
			limit = {
				OR = {
					has_government = theocracy_government
					is_landed = no
				}
				is_capable_adult = yes
				faith = scope:holder.faith
				OR = {
					scope:holder.faith = { has_doctrine = doctrine_clerical_marriage_allowed }
					is_married = no
				}
				NOT = {
					scope:holder.faith = { has_doctrine = doctrine_clerical_gender_male_only }
					is_female = yes
				}
				NOT = {
					scope:holder.faith = { has_doctrine = doctrine_clerical_gender_female_only }
					is_female = no
				}
			}
		}
	}
}
