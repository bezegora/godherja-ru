magic_legacy_1 = { # Magic for the Pure
	legacy = magic_legacy_track

	effect = {
		# Effect is applied in the magic_defines file
		custom_description_no_bullet = {
			text = magic_legacy_1_effect
		}
	}
	
	ai_chance = {
		value = 11
		if = {
			limit = {
				AND = {
					has_culture_group = culture_group:gh_marcher_group
					can_access_magic = no
				}
			}
			multiply = -10
		}
		if = {
			limit = {
				AND = {
					has_culture_group = culture_group:gh_marcher_group
					can_access_magic = yes
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				AND = {
					NOT = {
						has_culture_group = culture_group:gh_marcher_group
					}
					can_access_magic = yes
				}
			}
			multiply = 5
		}
		if = {
			limit = {
				has_culture_group = culture_group:gh_imperial_group
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_legacy_track_trigger = no
			}
			multiply = 0
		}
	}
}
