amsari_child_kill = {
	category = interaction_category_hostile

	greeting = positive

	auto_accept = yes

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			has_title = title:e_amsari
			title:e_amsari = {
				holder = {
					has_culture_group = culture_group:gh_zarasadari_group
				}
			}
		}
		scope:recipient = {
			is_child_of = scope:actor
			age < 16
		}
	}

	on_accept = {
		scope:recipient = {
			death = {
				death_reason = death_strangulation
				killer = scope:actor
			}
		}
		scope:actor.primary_heir = {
			add_opinion = {
				modifier = supportive_parent
				target = scope:actor
			}
			desc = happy_heir
		}
		scope:actor = {
			every_child = {
				limit = {
					NOT = {
						this = scope:actor.primary_heir
					}
				}
				add_opinion = { 
					modifier = supportive_parent
					target = scope:actor
				}
			}
			desc = happy_heir
		}
	}

	ai_targets = {
		ai_recipients = children
		max = 10
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_frequency = 72
	
	ai_potential = {
		NOT = {
			this = character:canian_1
		}
		any_child = {
			count >= 3
		}
	}
}
