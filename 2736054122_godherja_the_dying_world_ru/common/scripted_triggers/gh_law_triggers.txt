
can_keep_sjalvolki_succession_law_trigger = {
	trigger_if = {
		limit = {
			AND = {
				has_culture_group = culture_group:gh_sjalvolki_group
				has_religion = religion:sjalvolki_religion
				faith = {
					has_doctrine = tenet_sjalvolki_syncretism
				}
				has_realm_law = sjalvolki_succession_law
			}			
		}
		can_have_sjalvolki_succession_law_trigger = yes
	}
}

can_have_sjalvolki_succession_law_trigger = {
	custom_description = {
		text = succession_laws_must_have_hereditary_government_tooltip
		OR = {
			has_government = feudal_government
			has_government = tribal_government
			has_government = clan_government
		}
	}
	custom_description = {
		text = succession_laws_must_be_sjalvolki_tooltip
		OR = {
			has_culture_group = culture_group:gh_sjalvolki_group
			has_religion = religion:sjalvolki_religion
			faith = {
				has_doctrine = tenet_sjalvolki_syncretism
			}
		}
	}
}
