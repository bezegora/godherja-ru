	remove_variable = tmp_$MODE$
}

uncorrupt_adoxos_effect = {
	var:clique_title = {
		random_in_list = {
			variable = title_cliques
			limit = { var:clique_name = flag:adoxos }
			save_scope_as = adoxos_clique
		}
	}
	scope:adoxos_clique = {
		remove_guiding_principle_trait_effect = { TRAIT = sadistic }
		remove_guiding_principle_trait_effect = { TRAIT = wrathful }
		remove_guiding_principle_opposed_trait_effect = { TRAIT = compassionate }
		remove_guiding_principle_opposed_trait_effect = { TRAIT = calm }
		set_guiding_principle_trait_effect = { TRAIT = calm }
		set_guiding_principle_trait_effect = { TRAIT = zealous }
		set_guiding_principle_opposed_trait_effect = { TRAIT = impatient }
		set_guiding_principle_opposed_trait_effect = { TRAIT = fickle }
		
		
		remove_clique_ai_values_effect = { AI_VALUE = ai_compassion }
		remove_clique_ai_values_effect = { AI_VALUE = ai_honor }
		remove_clique_ai_values_effect = { AI_VALUE = ai_rationality }
		set_clique_ai_values_effect = { AI_VALUE = ai_honor 	MODIFIER = 0.1 }
		set_clique_ai_values_effect = { AI_VALUE = ai_zeal 	MODIFIER = 0.25 }
		
		remove_variable = clique_name
		set_variable = {
			name = clique_name
			value = flag:adoxos_pure
		}
	}
}

	every_ruler = {
		limit = {
			has_clique_government_trigger = yes
			highest_held_title_tier > tier_barony
			NOT = {
				any_held_title = {
					has_variable = clique_number
				}
			}
			OR = {
				is_independent_ruler = yes
				liege = { has_clique_government_trigger = no }
				NOT = { has_same_government = liege }
				AND = {
					liege = { 
						OR = {
							has_government = landed_order_government 
							has_government = theocracy_government
						}
					}
					NOT = { faith = liege.faith }
				}
			}
		}
		primary_title = { set_default_cliques_effect = yes }
	}
}
