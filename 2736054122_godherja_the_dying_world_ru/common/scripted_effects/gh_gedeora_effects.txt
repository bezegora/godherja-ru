apply_gedeora_secondary_effect_to_character = {
	if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_magic_surge } }
		change_trait_rank = {
			trait = magic_good
			rank = 1
		}
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_magic_drop } }
		change_trait_rank = {
			trait = magic_good
			rank = -1
		}
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_lunatic } }
		add_trait = lunatic_1
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_melancholy } }
		add_trait = depressed_1
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_impregnate } }
		if = {
			limit = { 
				is_pregnant = no
				is_female = yes
				primary_partner = {
					exists = this
				}
			}
			make_pregnant = {
				father = primary_partner
				number_of_children = 1
			}
			set_variable = { 
				name = carrying_gedeora_child
				months = 12	# Should auto-expire
			}
		}
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_barren } }
		if = {
			limit = { is_pregnant = yes }
			end_pregnancy = yes
		}
		add_trait = infertile
	}
	else_if = {
		limit = { scope:gedeora_epicenter = { has_variable = gedeora_death } }
		death = {
			death_reason = death_gedeora
		}
	}
}
