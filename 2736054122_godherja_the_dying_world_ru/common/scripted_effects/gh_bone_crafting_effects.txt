bone_artifact_craft_effect = {
	if = {
		limit = {
			scope:bone_level = 4
			}
		bone_artifact_craft_effect_backend = {ARTIFACT_TYPE = $ARTIFACT_TYPE$ ARTIFACT_RANK = 4}
	}
	else_if = {
		limit = {
			scope:bone_level = 3
			}
		bone_artifact_craft_effect_backend = {ARTIFACT_TYPE = $ARTIFACT_TYPE$ ARTIFACT_RANK = 3}
	}
	else_if = {
		limit = {
			scope:bone_level = 2
			}
		bone_artifact_craft_effect_backend = {ARTIFACT_TYPE = $ARTIFACT_TYPE$ ARTIFACT_RANK = 2}
	}
	else_if = {
		limit = {
			scope:bone_level = 1
			}
		bone_artifact_craft_effect_backend = {ARTIFACT_TYPE = $ARTIFACT_TYPE$ ARTIFACT_RANK = 1}
	}
}

bone_artifact_craft_effect_backend = {
	add_artifact = {ARTIFACT = $ARTIFACT_TYPE$_$ARTIFACT_RANK$}
}
