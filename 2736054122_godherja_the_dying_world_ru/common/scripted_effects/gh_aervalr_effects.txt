#Moves forward cycle, clamps down to 1 
next_aervalrian_cycle_effect = {
    if = {
        limit = { global_var:aervalrian_cycle = 8}
        set_global_variable = {
            name = aervalrian_cycle
            value = 1
        }
    }
    else = {
        change_global_variable = {
            name = aervalrian_cycle
            add = 1
        }
    }
}
