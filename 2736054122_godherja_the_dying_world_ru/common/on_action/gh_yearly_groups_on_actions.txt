legal_tradition_yearly_events = {
	trigger = {
		has_trait = lifestyle_jurist
		NOR = { 
			has_variable = had_recent_legal_tradition_event
			trigger_if = {
				limit = { has_variable = judicial_workload }
				var:judicial_workload = flag:low_workload
			}
		}
	}
	weight_multiplier = {
		base = 1
	}

	random_events = {
		chance_to_happen = 100
		
		#Legal Events
		100 = legal_tradition.0501 # Criminal Law
		100 = legal_tradition.0511 # Sentencing Policy
		100 = legal_tradition.0521 # Torts
		100 = legal_tradition.0531 # Financial Law
		100 = legal_tradition.0541 # Guild Law
		100 = legal_tradition.0551 # Religious Law
		100 = legal_tradition.0561 # Feudal Law
	}
	
	effect = {
		if = {
			limit = {
				trigger_if = {
					limit = { exists = var:judicial_workload }
					NOT = { var:judicial_workload = flag:high_workload }
				}
			}
			set_variable = {
				name = had_recent_legal_tradition_event
				days = 365
			}
		}
	}
}
