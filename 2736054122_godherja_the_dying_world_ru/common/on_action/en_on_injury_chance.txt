on_injury_nickname = 
{
	trigger = { prestige_level >= 2 }
	first_valid_on_action = 
	{
		on_scarred_nickname
		on_disfigured_nickname 
		on_maimed_nickname
		on_weak_nickname
	}
}

on_scarred_nickname = 
{
	trigger = { has_trait >= scarred }
	random_on_action = 
	{
		30 = 0		
		10 = on_three_fingers
		10 = on_two_fingers
		10 = on_the_scarred
		10 = on_the_bloodied
		10 = on_the_marked
		10 = on_the_stiched
		10 = on_the_fingerless
		10 = on_the_marred
		10 = on_the_carved
	}
}
