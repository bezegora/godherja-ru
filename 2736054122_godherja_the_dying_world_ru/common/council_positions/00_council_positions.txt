councillor_court_mage = {
	skill = learning
	
	name = councillor_court_mage

	tooltip = game_concept_court_mage_desc
	auto_fill = {
		liege = {
			can_have_court_magi_trigger = yes
		}
		highest_held_title_tier < tier_duchy
		can_access_magic = yes
	}

	valid_position = {
		can_have_court_magi_trigger = yes
	}

	valid_character = {
		trigger_if = {
			limit = { exists = liege }
			liege = {
				can_have_court_magi_trigger = yes
			}
		}
		can_access_magic = yes
	}

	modifier = {
		name = councillor_court_mage_modifier
		learning = 1
		monthly_prestige = 0.5
		monthly_learning_lifestyle_xp_gain_mult = 0.01
	}	

	on_get_position = {
		gave_minor_title_effect = yes 					
	}
	
	on_fired_from_position = {
		if = {
			limit = { is_powerful_vassal = no }
			remove_opinion = {
				modifier = gave_minor_title_normal_vassal_opinion
				target = liege
			}
		}
		else_if = {
			remove_opinion = {
				modifier = gave_minor_title_powerful_vassal_opinion
				target = liege
			}
		}
	}	
	
	portrait_animation = court_mage
}
