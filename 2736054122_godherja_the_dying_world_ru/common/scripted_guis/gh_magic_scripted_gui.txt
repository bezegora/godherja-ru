magic_counter_display = {
	scope = character
	
	is_shown = {
		OR = {
			has_trait_rank = {
				trait = magic_good
				rank > 0
			}
			has_trait_rank = {
				trait = magic_good
				rank > 1
			}
			has_trait_rank = {
				trait = magic_good
				rank > 2
			}
			has_trait_rank = {
				trait = magic_good
				rank > 3
			}
			has_trait_rank = {
				trait = magic_good
				rank > 4
			}
		}
	}
}
