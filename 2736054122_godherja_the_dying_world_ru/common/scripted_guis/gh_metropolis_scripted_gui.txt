	}

build_district_scripted_gui = {
	scope = province
	
	is_shown = {
	
	}
	
	is_valid = {
		exists = scope:county_owner
		scope:county_owner = { 
			gold >= 500
		}
	}
	
	effect = {
		scope:county_owner = {
			remove_short_term_gold = 500
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = maritime_district_holding } }
			set_holding_type = maritime_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = military_district_holding } }
			set_holding_type = military_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = palatial_district_holding } }
			set_holding_type = palatial_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = trade_district_holding } }
			set_holding_type = trade_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = temple_district_holding } }
			set_holding_type = temple_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = workshop_district_holding } }
			set_holding_type = workshop_district_holding
		}
		if = {
			limit = { holding_is_type_from_flag_trigger = { FLAG_SCOPE = scope:holding_type_scope FLAG = urban_district_holding } }
			set_holding_type = urban_district_holding
		}
	}
}
