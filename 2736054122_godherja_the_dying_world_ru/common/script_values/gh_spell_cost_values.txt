flagellate_cost = {
	value = 0
	if = {	# The closer you beat yourself to the brink of death, the more magic you get
		limit = { 
			NOT = { 
				has_trait_rank = {
					trait = wounded
					rank >= 1
				}
			} 
		}
		add = tier_2_gen
	}
	else_if = {
		limit = { 
			has_trait_rank = {
				trait = wounded
				rank = 1
			}
		}
		add = tier_3_gen
	}
	else_if = {
		limit = { 
			has_trait_rank = {
				trait = wounded
				rank = 2
			}
		}
		add = tier_4_gen
	}
	if = {
		limit = { 
			has_perk = livingmagic_perk8_tree1
		}
		multiply = 1.5
	}
}

sacrifice_plants_gen_province_value = {
	value = 0
	if = {
			limit = {
				OR = {
				terrain = desert
				terrain = desert_mountains
				terrain = drylands
				}
			}
		add = 1
	}
	if = {
			limit = {
				OR = {
				terrain = mountains
				terrain = marcher_mountains
				terrain = wetlands
				terrain = marcher_wetlands
				terrain = steppe
				terrain = hills
				terrain = marcher_hills
				}
			}
		add = 2
	}
	if = {
			limit = {
				OR = {
				terrain = plains
				terrain = marcher_plains
				terrain = farmlands
				terrain = marcher_farmlands
				terrain = oasis
				terrain = floodplains
				terrain = marcher_floodplains
				}
			}
		add = 3
	}
	if = {
			limit = {
				OR = {
				terrain = taiga
				terrain = forest
				terrain = marcher_forest
				terrain = farmlands
				terrain = marcher_farmlands
				terrain = jungle
				}
			}
		add = 4
	}
}
