
religion_county_number_value = {
	value = 0
	every_faith = {
		add = num_county_followers
	}
}

faith_county_number_percentage_value = {
	value = 0
	add = num_county_followers
	if = {
		limit = { religion.religion_county_number_value > 0 }
		divide = religion.religion_county_number_value
		multiply = 100
	}
	else = {
		value = 0
	}
}
