faith_hostility_level_conversion_malus = {
	value = 1
	if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:councillor.faith
					value >= faith_hostile_level
				}
			}
		}
		add = 2
	}
	if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:councillor.faith
					value >= faith_evil_level
				}
			}
		}
		add = 3
	}
}

duchy_same_to_target_kingdom_neighboring_difference_value = {
	value = 0
	every_title_to_title_neighboring_duchy = {
		limit = { kingdom = prev.kingdom }
		add = 1
	}
	if = {
		limit = { 
			exists = scope:target_kingdom
		}
		every_title_to_title_neighboring_duchy = {
			limit = { kingdom = scope:target_kingdom }
			subtract = 1
		}
	}
}
