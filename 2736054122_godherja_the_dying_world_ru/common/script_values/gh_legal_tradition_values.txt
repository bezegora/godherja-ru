		add = 20
	}
	if = {
		limit = { has_trait_with_flag = character_judicial_power_25 }
		add = 25
	}
	add = judicial_power_from_legal_tradition_value
	if = {
		limit = { has_trait_with_flag = character_judicial_power_factor_10 }
		multiply = 1.1
	}
	if = {
		limit = { has_trait_with_flag = character_judicial_power_factor_25 }
		multiply = 1.25
	}
	if = {
		limit = { has_trait_with_flag = character_judicial_power_factor_50 }
		multiply = 1.5
	}
	if = {
		limit = { has_trait_with_flag = character_judicial_power_factor_100 }
		multiply = 2
	}
	if = {
		limit = { has_trait_with_flag = character_judicial_power_factor_200 }
		multiply = 3
	}
	if = {
		limit = {
			legal_tradition_has_flag_trigger = { FLAG = minor_legal_tradition }
		}
		multiply = 0.5
	}
	if = {
		limit = {
			legal_tradition_has_flag_trigger = { FLAG = radicalism_gives_judicial_power }
			has_variable = radicalism
		}
		add = {
			value = var:radicalism
			multiply = 0.2
			max = 20
		}
	}
	if = {
		limit = {
			exists = dynasty
			dynasty = { has_dynasty_perk = wardenite_legacy_4 }
		}
		multiply = 1.2
	}
}

# Amount of progress points required to progress to the next legal tradition level
legal_tradition_next_level_progress_requirement_value = {
	value = 1000	# Base level
	if = {
		limit = { has_variable = legal_tradition_level }
		if = {
			limit = { var:legal_tradition_level = 0 }
			multiply = 2
		}
		else_if = {
			limit = { var:legal_tradition_level = 1 }
			multiply = 1
		}
		else_if = {
			limit = { var:legal_tradition_level = 2 }
			multiply = 2.5
		}
		else_if = {
			limit = { var:legal_tradition_level = 3 }
			multiply = 5
		}
		else_if = {
			limit = { var:legal_tradition_level = 4 }
			multiply = 10
		}
	}
}

# Utility script value that divides the percentile value of your patronage investment into a 0-1 value
legal_tradition_patronage_investment_amount_decimal_value = {
	value = 0
	if = {
		limit = { has_variable = character_legal_tradition_patronage_level }
		add = var:character_legal_tradition_patronage_level
		divide = 100
	}
}

legal_tradition_patronage_investment_amount_inverted_decimal_value = {
	value = 1
	subtract = legal_tradition_patronage_investment_amount_decimal_value
}

# Amount of progress gained by a character's patronage
legal_tradition_patronage_investment_character_total_value = {
	value = monthly_character_income
	multiply = legal_tradition_patronage_investment_amount_decimal_value
	divide = legal_tradition_patronage_investment_amount_inverted_decimal_value
}

# Projected patronage cost increase
legal_tradition_projected_patronage_increase_cost_value = {
	value = monthly_character_income

		value = 0
		if = {
			limit = { has_variable = character_legal_tradition_patronage_level }
			add = var:character_legal_tradition_patronage_level
		}
		add = 10
		divide = 100
	}
}

# Amount of progress gained monthly, total
