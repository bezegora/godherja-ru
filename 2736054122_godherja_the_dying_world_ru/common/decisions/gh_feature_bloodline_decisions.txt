# Found Diplomacy Bloodline
# by 𝑽𝒂𝒓𝒍𝒂𝒕𝒓𝒂
found_diplomacy_bloodline_decision = {
    picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"

    desc = found_diplomacy_bloodline_decision_desc
    selection_tooltip = found_diplomacy_bloodline_decision_tooltip

    is_shown = {
        is_adult = yes
        is_ruler = yes
        NOR = { 
			exists = global_var:bloodline_diplomacy_skill_founder
			has_variable = already_founded_bloodline
		}
        OR = {
            diplomacy >= 25
            AND = {
                diplomacy >= 20
                prestige_level >= 4
            }
        }
    }

    is_valid_showing_failures_only = {
        diplomacy >= 25
        prestige_level >= 4
    }

    effect = {
        #custom_tooltip = found_diplomacy_bloodline_decision_tt
        #trigger_event = bloodline.001
        bloodline_set_founder = { TRAIT = diplomacy_skill }
    }

    ai_check_interval = 96

    ai_potential = {
        always = yes
    }

    ai_will_do = {
        base = 10
    }
}
