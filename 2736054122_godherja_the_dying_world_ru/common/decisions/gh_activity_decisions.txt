	picture = "gfx/interface/illustrations/decisions/decision_activity.dds"
	ai_check_interval = 60
	

	is_shown = {
		is_landed = yes
		OR = {
			any_realm_province = {
				OR = {
					has_province_modifier = province_adventure_creepy_graveyard_modifier
					has_province_modifier = province_adventure_ancient_ruin_modifier
					has_province_modifier = province_adventure_Forboding_Temple_modifier
					has_province_modifier = province_adventure_Abandoned_Dungeon_modifier
					has_province_modifier = province_adventure_Mysterious_Cave_modifier
				}
			}
		}
	}


	is_valid_showing_failures_only = {
		is_available_adult = yes
		is_at_war = no
	}
	
	cost = {
		gold = { add = standard_activity_cost }
	}

	cooldown = { months = 36 }

	effect = {
		trigger_event = adventure.100
		custom_tooltip = needadventurelocation
	}
}
