# Reform the shattered Empire
# by Netgearghost
reform_the_shattered_empire = {
	picture = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	major = yes
	ai_check_interval = 12
	desc = reform_the_shattered_empire_desc

	is_shown = {
		has_culture_group = culture_group:gh_zarasadari_group
		has_global_variable = amsari_civil_war
		has_variable = amsari_war1
		NOT = {
			has_title = title:e_amsari
		}
	}

	is_valid = {
		has_culture_group = culture_group:gh_zarasadari_group
		any_in_global_list = {
			variable = amsari_war
			count <= 1
		}
		has_variable = amsari_war1
		has_title = title:c_tchouran
		is_landed = yes
		is_available_at_peace_adult = yes
		NOT = {
			has_variable = amsari_neutral
		}
	}

	effect = {
		custom_tooltip = reform_the_shattered_empire_tt	
		get_title = title:e_amsari
		trigger_event = amsari.0008
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

	picture = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	major = yes
	ai_check_interval = 24
	desc = hire_marcher_mercenaries_desc

	is_shown = {
		has_culture_group = culture_group:gh_zarasadari_group
		has_global_variable = amsari_civil_war
		has_variable = amsari_war1
	}

	is_valid = {
		has_culture_group = culture_group:gh_zarasadari_group
		has_variable = amsari_war1
	}

	cost = {
		gold = 100
	}

	cooldown = { days = 1460 }

	effect = {
		custom_tooltip = hire_marcher_mercenaries_tt	
		
		spawn_army = {
			uses_supply = no
			inheritable = no
			name = marcher_mercenaries
			levies = {
				value = 300
			}
			men_at_arms = {
				type = marcher_anti_mage
				stacks = 3
			}
			location = capital_province
		}
		title:d_marcher_amsari_republic = {
			holder = { add_gold = 100 }
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

	desc = the_hiklahn_needs_change_desc

	is_shown = {
		has_culture_group = culture_group:gh_zarasadari_group
		NOT = {
			has_global_variable = amsari_civil_war
			has_variable = amsari_war1
			has_trait = bloodline_the_prophet
		}
		has_title = title:e_amsari
	}

	is_valid = {
		piety_level >= 5
		prestige_level >= 3
		has_title = title:e_amsari
	}

	effect = {
		add_prestige = -750
		add_piety = -750
		set_global_variable = {
			name = reformed_amsari
			value = this
		}
	}
}
