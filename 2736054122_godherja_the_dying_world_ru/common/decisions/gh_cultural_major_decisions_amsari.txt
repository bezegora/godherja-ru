reform_the_shattered_empire = {
	picture = "gfx/interface/illustrations/decisions/decision_aversaria.dds"
	major = yes
	ai_check_interval = 12
	desc = reform_the_shattered_empire

	is_shown = {
		has_culture_group = culture_group:gh_zarasadari_group
		has_global_variable = amsari_civil_war
	}

	is_valid = {
		has_culture_group = culture_group:gh_zarasadari_group
		any_in_global_list = {
			variable = amsari_war
			count <= 1
		}
		is_available_at_peace_adult = yes
	}

	effect = {
		custom_tooltip = reform_the_shattered_empire_tt	
		
		get_title = e_amsari
		trigger_event = amsari.0008
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
