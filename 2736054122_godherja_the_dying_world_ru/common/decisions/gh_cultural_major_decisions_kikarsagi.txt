kikarsagi_kill_aversarians_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	ai_check_interval = 120
	desc = kikarsagi_kill_aversarians_decision_desc

	is_shown = {
		OR = {
			any_vassal_or_below = {
				culture_group = culture_group:gh_imperial_group
			}
			any_prisoner = {
				culture_group = culture_group:gh_imperial_group
			}
			any_held_title = {
				trigger_if = {
					limit = { tier = tier_county }
					culture_group = culture_group:gh_imperial_group
				}
				trigger_else = { always = no }
			}
		}
		culture_group = culture_group:gh_kikarsag_group
		faith = faith:aversarian_cumani
	}

	is_valid = {
		is_at_war = no
	}

	effect = {
	    add_trait = gh_chosen_man
		create_title_and_vassal_change = {
			type = revoked
			save_scope_as = change
		}

		every_vassal_or_below = {
			limit = {
				culture_group = culture_group:gh_imperial_group
			}
			prev = {
				imprison = {
					target = prev
				}
			}
			every_held_title = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
		}

		resolve_title_and_vassal_change = scope:change

		every_held_title = {
			limit = {
				culture_group = culture_group:gh_imperial_group
				tier = tier_county
			}
			
			set_county_culture = root.culture
			set_county_faith = root.faith

			root = {
				add_prestige = 50
				add_piety = 50
			}
		}

		every_prisoner = {
			limit = {
				culture_group = culture_group:gh_imperial_group
			}
			death = {
				death_reason = death_execution
				killer = root
			}

			IF = {
				limit = {
					highest_held_title_tier = tier_empire
				}
				root = {
					add_prestige = 200
					add_piety = 50
				}
			}
			ELSE_IF = {
				limit = {
					highest_held_title_tier = tier_kingdom
				}
				root = {
					add_prestige = 100
					add_piety = 40
				}
			}
			ELSE_IF = {
				limit = {
					highest_held_title_tier = tier_duchy
				}
				root = {
					add_prestige = 25
					add_piety = 30
				}
			}
			ELSE_IF = {
				limit = {
					highest_held_title_tier = tier_county
				}
				root = {
					add_prestige = 10
					add_piety = 20
				}
			}
			ELSE_IF = {
				limit = {
					highest_held_title_tier = tier_barony
				}
				root = {
					add_prestige = 5
					add_piety = 10
				}
			}
			ELSE = {
				root = {
					add_piety = 10
				}
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
