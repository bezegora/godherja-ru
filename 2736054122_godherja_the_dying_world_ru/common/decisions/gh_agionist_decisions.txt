blind_watcher_reform_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
	major = yes
	ai_check_interval = 5
	desc = blind_watcher_reform_decision_desc
	
	is_shown = {
		has_faith = faith:dikaynos
		has_government = landed_order_government
		has_variable = character_clique
		OR = {
			is_ai = no
			var:character_clique = {
				var:clique_name = flag:watcher_defenders
			}
		}
		faith = { has_doctrine = tenet_blind_justice }
	}
	
	is_valid = {
		custom_description = {
			text = blind_watcher_reform_correct_clique_desc
			var:character_clique = {
				var:clique_name = flag:watcher_defenders
			}
		}
		has_title = title:k_blind_watcher
		prestige >= 1000
		custom_description = {
			text = blind_watcher_reform_valid_desc
			character_clique_power_percentage >= 60
		}
		custom_description = {
			text = landed_order_reform_no_powerful_vassal_must_oppose_desc
			NOT = {
				any_vassal = {
					is_powerful_vassal = yes
					opposes_succession_law_change_trigger = yes
				}
			}
		}
	}
	
	effect = {
		add_piety = 1000
		add_prestige = -1000
		faith:dikaynos = {
			remove_doctrine = tenet_blind_justice
			add_doctrine = tenet_true_blind_justice
			hidden_effect = {
				remove_doctrine = doctrine_pluralism_fundamentalist
			}
			add_doctrine = doctrine_pluralism_righteous
		}
		uncorrupt_interpreters_effect = yes
		uncorrupt_swords_effect = yes
	}
}
