refound_east_mamur_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
	major = yes

	ai_check_interval = 120

	desc = refound_east_mamur_decision_desc
	selection_tooltip = refound_east_mamur_decision_tooltip

	is_shown = {
		#Standard filter checks.
		is_landed = yes
		exists = dynasty
		OR = {
			has_government = tribal_government
			has_government = feudal_government
			has_government = clan_government
		}
		#Must be Wardenite.
		has_religion = religion:sarradonian_religion
		#Must be in the vague region.
		any_held_title = {
			tier = tier_county
			OR = {
				empire = title:e_kekhenemhat
				empire = title:e_uhtash
				empire = title:e_farhadd
			}
		}
		#Can't already be a Mamur.
		NOT = {
			any_held_title = {
				has_variable = attached_high_court
			}
		}
		#The Eastern Mamuramat mustn't already exist.
		title:e_eastern_mamuramat = { is_title_created = no }
		#Can't be an Idealist
		NOT = { faith = { has_doctrine_parameter = no_mamurs_recognized } }
		#If the faith recognizes only one Mamur, make sure that it doesn't already have an appropriate one, since we can't create new faiths currently to represent schisms
		trigger_if = {
			limit = { faith = { has_doctrine_parameter = one_mamur_recognized } }
			NOT = {
				faith = {
					save_temporary_scope_as = character_faith
					any_in_list = {
						variable = recognized_high_court_list
						has_variable = high_court_mamuramat
						exists = var:high_court_mamuramat.holder
						var:high_court_mamuramat.holder = { has_faith = scope:character_faith }
					}
				}
			}
		}
	}

	is_valid = {
		#Should have a decent piety level.
		piety_level >= very_high_piety_level
		#Rules Kekhenemhat
		has_title = title:e_kekhenemhat
		#Rules Kundahar
		has_title = title:c_kudahhar
	}

	is_valid_showing_failures_only = {
		is_alive = yes
		is_in_an_activity = no
		#Has to be around for the ceremony.
		is_imprisoned = no
		has_contagious_deadly_disease_trigger = no
	}

	effect = {
		save_scope_as = founder
		found_mamuramat_effect = { MAMURAMAT_TITLE = title:e_eastern_mamuramat }
	}

	cost = {
		gold = 200
		piety = {
			value = {
				value = 2000
				#Traditionalists get a massive discount
				if = {
					limit = { faith = { has_doctrine_parameter = four_traditional_mamurs_recognized } }
					multiply = 0.25
				}
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
