
EvidenceStrength = {
	type = character
	
	text = {
		localization_key = nonexistent_evidence
		trigger = {
			exists = var:evidence_level
			var:evidence_level <= 20
		}
	}
	
	text = {
		localization_key = poor_evidence
		trigger = {
			exists = var:evidence_level
			var:evidence_level > 20
			var:evidence_level <= 40
		}
	}
	
	text = {
		localization_key = uncertain_evidence
		trigger = {
			exists = var:evidence_level
			var:evidence_level > 40
			var:evidence_level <= 60
		}
	}
	
	text = {
		localization_key = strong_evidence
		trigger = {
			exists = var:evidence_level
			var:evidence_level > 60
			var:evidence_level <= 80
		}
	}
	
	text = {
		localization_key = very_strong_evidence
		trigger = {
			exists = var:evidence_level
			var:evidence_level > 80
		}
	}
	
	text = {
		localization_key = uncertain_evidence
		fallback = yes
	}
}
