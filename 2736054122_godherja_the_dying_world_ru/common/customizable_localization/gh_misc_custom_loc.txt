CommerceRaidingCompetitionExists = {
	text = {
		trigger = {
			save_temporary_scope_as = pirate_strait_info_county
			any_in_list = {
				variable = strait_county_list
				has_government = pirate_government
				save_temporary_scope_as = first_pirate_in_strait
				scope:pirate_strait_info_county = {
					NOT = {
						any_in_list = {
							variable = strait_county_list
							has_government = pirate_government
							NOT = { top_liege = scope:first_pirate_in_strait.top_liege }
						}
					}
				}
			}
		}
		localization_key = commerce_raiding_competition_exists
	}

	text = {
		localization_key = commerce_raiding_competition_does_not_exist
	}
}
